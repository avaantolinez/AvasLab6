decoded_password = ""
encoded_password = ""


def password_encoder(digits):
    global encoded_password
    global decoded_password
    decoded_password = digits
    result = ""
    for i in range(len(digits)):
        addition = int(digits[i]) + 3
        if addition > 9:
            addition -= 10
        result += str(addition)
    encoded_password = result
